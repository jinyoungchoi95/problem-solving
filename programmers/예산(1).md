import java.util.*;

class Solution {
    public int solution(int[] d, int budget) {
        int answer = 0;
        Arrays.sort(d);
        int index = 0;
        while(budget > 0 && index < d.length) {
            answer++;
            budget -= d[index++];
        }
        
        if (budget < 0) {
            answer --;
        }
        return answer;
    }
}
