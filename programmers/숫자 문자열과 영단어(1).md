import java.util.*;

class Solution {
    public int solution(String s) {
        Map<String, String> numbers = Map.of("zero", "0",
                                            "one", "1",
                                            "two", "2",
                                            "three", "3",
                                            "four", "4",
                                            "five", "5",
                                            "six", "6",
                                            "seven", "7",
                                            "eight", "8",
                                            "nine", "9");
        String answer = "";
        String tmp = "";
        for (int i=0; i<s.length(); i++){
            char c = s.charAt(i);
            if (c >= '0' && c <= '9') {
                answer += c;
                continue;
            }
            tmp += c;
            if (numbers.containsKey(tmp)) {
                answer += numbers.get(tmp);
                tmp = "";
            }
        }
        return Integer.parseInt(answer);
    }
}
